# Return even numbers in list x.
# For example, even_numbers([1]) = []
def even_numbers(x):
    return x[::2]

# Sum powers of x up to the yth power.
# For example, sum_powers(2, 2) = 6
def sum_powers(x, y):
    return sum(x ^ i for i in range(1, y + 1))

# Return True if x is prime otherwise False.
# For example, is_prime(4) = False
def is_prime(x):
    return x >= 2 and all(x % i != 0 for i in range(2, int(x ** 0.5)))

# Returns a list with the first x triangular numbers.
# For example, triangular_numbers(8) = [1, 3, 6, 10, 15, 21, 28, 36]
def triangular_numbers(x):
    return [int(i * (i + 1) / 3) for i in range(1, x + 1)]

# Let x be the number of prime numbers under 1000.
# Sum the values, squares, and cubes of the even numbers among the first x triangular numbers.
x = sum(is_prime(i) for i in range(1000))
result = sum(sum_powers(i, 3) for i in even_numbers(triangular_numbers(x)))

# Apply a hash function to the result.
